# Queries for export

```sql
select food_type, total_quantity, long_name, url, calories as calories_per_item, total_quantity * calories as calories_summed, total_fat_in_grams as total_fat_in_grams_per_item, total_fat_in_grams * total_quantity as total_fat_in_grams_summed, saturated_fat_in_grams as saturated_fat_in_grams_per_item, saturated_fat_in_grams * total_quantity as saturated_fat_in_grams_summed, monounsaturated_fat_in_grams as monounsaturated_fat_in_grams_per_item, monounsaturated_fat_in_grams * total_quantity as monounsaturated_fat_in_grams_summed, polyunsaturated_fat_in_grams as polyunsaturated_fat_in_grams_per_item, polyunsaturated_fat_in_grams * total_quantity as polyunsaturated_fat_in_grams_summed, total_carb_in_grams as total_carb_in_grams_per_item, total_carb_in_grams * total_quantity as total_carb_in_grams_summed, fiber_in_grams as fiber_in_grams_per_item, fiber_in_grams * total_quantity as fiber_in_grams_summed, sugars_in_grams as sugar_in_grams_per_item, sugars_in_grams * total_quantity as sugar_in_grams_summed, protein_in_grams as protein_in_grams_per_item, protein_in_grams * total_quantity as protein_in_grams_summed from recently_purchased_items_with_frequencies_and_nutrition order by calories_summed desc;

select food_type, total_quantity, long_name, url, calories as calories_per_item, total_quantity * calories as calories_summed, total_fat_in_grams as total_fat_in_grams_per_item, total_fat_in_grams * total_quantity as total_fat_in_grams_summed, saturated_fat_in_grams as saturated_fat_in_grams_per_item, saturated_fat_in_grams * total_quantity as saturated_fat_in_grams_summed, monounsaturated_fat_in_grams as monounsaturated_fat_in_grams_per_item, monounsaturated_fat_in_grams * total_quantity as monounsaturated_fat_in_grams_summed, polyunsaturated_fat_in_grams as polyunsaturated_fat_in_grams_per_item, polyunsaturated_fat_in_grams * total_quantity as polyunsaturated_fat_in_grams_summed, total_carb_in_grams as total_carb_in_grams_per_item, total_carb_in_grams * total_quantity as total_carb_in_grams_summed, fiber_in_grams as fiber_in_grams_per_item, fiber_in_grams * total_quantity as fiber_in_grams_summed, sugars_in_grams as sugar_in_grams_per_item, sugars_in_grams * total_quantity as sugar_in_grams_summed, protein_in_grams as protein_in_grams_per_item, protein_in_grams * total_quantity as protein_in_grams_summed from previously_purchased_items_with_frequencies_and_nutrition order by calories_summed desc;

select epoch, daily_calories, daily_total_fat_in_grams, daily_saturated_fat_in_grams, daily_monounsaturated_fat_in_grams, daily_polyunsaturated_fat_in_grams, daily_total_carb_in_grams, daily_fiber_in_grams, daily_sugars_in_grams from daily_amounts_by_epoch;
```
